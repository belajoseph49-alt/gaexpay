   199→  async function handleSignup(e: React.FormEvent) {
   200→    e.preventDefault();
   201→    if (loading) return;
   202→
   203→    if (!firstName.trim() || !lastName.trim()) return toast.error("Please enter your name");
   204→    if (!EMAIL_RE.test(email)) return toast.error("Please enter a valid email");
   205→    if (phone.replace(/\D/g, "").length < 7) return toast.error("Please enter a valid phone number");
   206→    if (password.length < 8) return toast.error("Password must be at least 8 characters");
   207→    if (!/[A-Za-z]/.test(password) || !/[0-9]/.test(password)) {
   208→      return toast.error("Password must contain a letter and a number");
   209→    }
   210→    if (password !== confirmPassword) return toast.error("Passwords do not match");
   211→    if (accountType === "business" && companyName.trim().length < 2) {
   212→      return toast.error("Please enter your company name");
   213→    }
   214→
   215→    setLoading(true);
   216→    try {
   217→      const res = await fetch("/api/auth/signup", {
   218→        method: "POST",
   219→        headers: { "Content-Type": "application/json" },
   220→        credentials: "include",
   221→        body: JSON.stringify({
   222→          firstName, lastName, email, phone, password,
   223→          country: country || undefined,
   224→          accountType,
   225→          companyName: accountType === "business" ? companyName : undefined,
   226→        }),
   227→      });
   228→      const data = await res.json();
   229→      if (!res.ok) throw new Error(data?.error || "Signup failed");
   230→      setAuthed();
   231→      toast.success(`Welcome to GaexPay, ${firstName}! 🎉`);
   232→      onSuccess();
   233→    } catch (err) {
   234→      toast.error(err instanceof Error ? err.message : "Signup failed");
   235→    } finally {
   236→      setLoading(false);
   237→    }
   238→  }
