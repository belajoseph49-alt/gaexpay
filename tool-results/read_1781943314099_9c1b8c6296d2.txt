     1→/**
     2→ * src/lib/country-services.ts
     3→ *
     4→ * Comprehensive database of REAL service providers per African country,
     5→ * organized by category. Used by GaexPay's bills/merchant directory so
     6→ * users see the actual providers in their country/city.
     7→ *
     8→ * Categories:
     9→ *   - transport         (ride-hailing, bus, intercity-bus, train, ferry)
    10→ *   - restaurants       (fast-food, restaurant, café, chain)
    11→ *   - supermarkets      (supermarket, hypermarket, convenience)
    12→ *   - hospitals         (public, private, teaching, clinic)
    13→ *   - pharmacies        (pharmacy, drug-store)
    14→ *   - internetProviders (mobile-data, broadband, fiber, satellite)
    15→ *   - universities      (public, private)
    16→ *   - fuelStations      (fuel)
    17→ *   - electricity       (electricity)
    18→ *   - water             (water)
    19→ *   - tv                (satellite-tv, cable-tv, streaming)
    20→ *   - insurance         (insurance)
    21→ *   - government        (tax, social-security, customs)
    22→ *
    23→ * Pricing is approximate (public list prices at time of writing) and is
    24→ * intended for representative display in the GaexPay demo. Always confirm
    25→ * the current price with the provider before paying.
    26→ */
    27→
    28→// ----- Shared types -------------------------------------------------------
    29→
    30→export interface CityProvider {
    31→  name: string;
    32→  type: string;
    33→  cities: string[];
    34→}
    35→
    36→export interface PlanProvider {
    37→  name: string;
    38→  type: string;
    39→  plans: { name: string; price: number; currency: string }[];
    40→}
    41→
    42→export interface UniversityProvider {
    43→  name: string;
    44→  type: string;
    45→  city: string;
    46→  tuitionRange: string;
    47→}
    48→
    49→export interface CoverageProvider {
    50→  name: string;
    51→  type: string;
    52→  coverage: string;
    53→}
    54→
    55→export interface PackageProvider {
    56→  name: string;
    57→  type: string;
    58→  packages: { name: string; price: number; currency: string }[];
    59→}
    60→
    61→export interface ServiceListProvider {
    62→  name: string;
    63→  type: string;
    64→  services: string[];
    65→}
    66→
