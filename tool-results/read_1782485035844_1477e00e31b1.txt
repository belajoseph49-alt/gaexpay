     1→/**
     2→ * src/lib/i18n/translations.ts
     3→ *
     4→ * GaexPay internationalisation — lightweight dictionary-based i18n.
     5→ *
     6→ * 12 supported languages (one RTL: Arabic):
     7→ *   en, fr, ru, zh, ar, es, de, ew, ff, sw, ln, ha
     8→ *
     9→ * Each `Record<string, string>` uses dotted keys, e.g. `nav.dashboard`.
    10→ * `{param}` placeholders are interpolated by `useTranslation().t()`.
    11→ *
    12→ * NOTE: For African languages (Ewondo, Fufulde, Lingala, Hausa) the
    13→ * translations below are the best-quality consensus renderings. Where a
    14→ * given fintech concept has no native term, we keep the French/English
    15→ * loanword commonly used in the region's mobile-money UIs.
    16→ */
    17→
    18→export type LanguageCode =
    19→  | "en" | "fr" | "ru" | "zh" | "ar" | "es" | "de"
    20→  | "ew" | "ff" | "sw" | "ln" | "ha";
    21→
    22→export interface LanguageMeta {
    23→  code: LanguageCode;
    24→  name: string;        // English name
    25→  nativeName: string;  // Endonym
    26→  flag: string;
    27→  rtl?: boolean;
    28→}
